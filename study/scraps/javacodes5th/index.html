১। আয়তক্ষেত্রের ক্ষেত্রফল নির্ণয়ের একটি জাভা প্রোগ্রাম লেখ?

import java.util.Scanner; // স্ক্যানার ক্লাস ইম্পোর্ট করা হয়েছে ব্যবহারকারীর ইনপুটের জন্য

public class RA {

    public static void main(String[] args) {

        Scanner s = new Scanner(System.in); // স্ক্যানার অবজেক্ট তৈরি করা হয়েছে

        System.out.print("দৈর্ঘ্য: "); // দৈর্ঘ্য ইনপুট নিতে বলা হচ্ছে
        double l = s.nextDouble(); // দৈর্ঘ্য পড়া হচ্ছে

        System.out.print("প্রস্থ: "); // প্রস্থ ইনপুট নিতে বলা হচ্ছে
        double w = s.nextDouble(); // প্রস্থ পড়া হচ্ছে

        double a = l * w; // ক্ষেত্রফল গণনা করা হচ্ছে

        System.out.println("ক্ষেত্রফল: " + a); // ক্ষেত্রফল প্রিন্ট করা হচ্ছে

        s.close(); // স্ক্যানার বন্ধ করা হচ্ছে
    }
}

২। Area নামে একটি ক্লাস তৈরি করে বৃত্তের ক্ষেত্রফল নির্ণয়ের একটি জাভা প্রোগ্রাম লেখ?

import java.util.Scanner; // স্ক্যানার ক্লাস ইম্পোর্ট করা হয়েছে

public class Area {

    public static void main(String[] args) {

        Scanner s = new Scanner(System.in); // স্ক্যানার অবজেক্ট তৈরি করা হয়েছে

        System.out.print("ব্যাসার্ধ প্রবেশ করান: "); // ব্যাসার্ধ ইনপুট নিতে বলা হচ্ছে
        double r = s.nextDouble(); // ব্যাসার্ধ পড়া হচ্ছে

        double a = Math.PI * r * r; // ক্ষেত্রফল গণনা করা হচ্ছে

        System.out.println("বৃত্তের ক্ষেত্রফল: " + a); // ক্ষেত্রফল প্রিন্ট করা হচ্ছে

        s.close(); // স্ক্যানার বন্ধ করা হচ্ছে
    }
}

৩। কনস্ট্রাক্টর মেথড ব্যবহার করে একটি জাভা প্রোগ্রাম লেখ?

class Student {

    int Roll;
    String Name;

    Student() { // কনস্ট্রাক্টর মেথড
        Roll = 56; // রোল সেট করা হচ্ছে
        Name = "Rifat"; // নাম সেট করা হচ্ছে
    }

    void showInfo() { // তথ্য দেখানোর মেথড
        System.out.println("Roll: " + Roll);
        System.out.println("Name: " + Name);
    }
}

public class ConstructorExample {

    public static void main(String[] args) {

        Student s = new Student(); // স্টুডেন্ট অবজেক্ট তৈরি করা হচ্ছে, কনস্ট্রাক্টর কল হবে
        s.showInfo(); // তথ্য দেখানো হচ্ছে
    }
}

৪। মেথড ওভারলোডিং পদ্ধতি ব্যবহার করে একটি জাভা প্রোগ্রাম লেখ?

class Rifat {

    int Rifat(int x, int y) { // দুটি পূর্ণসংখ্যার জন্য মেথড
        int z = x + y;
        System.out.println("Result: " + z);
        return 0;
    }

    float Rifat(float a, float b) { // দুটি দশমিক সংখ্যার জন্য মেথড
        float c = a + b;
        System.out.println("Result: " + c);
        return 0;
    }

    int Rifat(int x) { // একটি পূর্ণসংখ্যার জন্য মেথড
        System.out.println(x);
        return 0;
    }
}

public class Overload {

    public static void main(String[] args) {

        Rifat s = new Rifat();
        s.Rifat(56.57f, 757f); // ফ্লোট মেথড কল
        s.Rifat(10, 20); // ইন্টিজার মেথড কল
    }
}

৫। সুপার ক্লাস ও সাব ক্লাস ব্যবহার করে একটি জাভা প্রোগ্রাম লেখ?

class Animal { // সুপার ক্লাস
    void Sound() { // সাউন্ড মেথড
        System.out.println("Animal Sound"); // প্রাণীর শব্দ প্রিন্ট করা হচ্ছে
    }
}

class Dog extends Animal { // সাব ক্লাস
    void bark() { // বার্ক মেথড
        System.out.println("Dog bark"); // কুকুরের ডাক প্রিন্ট করা হচ্ছে
    }
}

public class Main {
    public static void main(String[] args) {
        Dog d = new Dog(); // Dog অবজেক্ট তৈরি করা হচ্ছে
        d.Sound(); // সুপার ক্লাসের মেথড কল করা হচ্ছে
        d.bark(); // সাব ক্লাসের মেথড কল করা হচ্ছে
    }
}
